Accent Colors

Section titled “Accent Colors”
/* Dark mode (default) */
:root {
  --sl-color-accent-low: color-mix(in oklch, #fb923c 15%, #030712);
  --sl-color-accent: #fb923c;
  --sl-color-accent-high: #fb923c;
}

/* Light mode */
:root[data-theme="light"] {
  --sl-color-accent-low: color-mix(in oklch, #c2410c 10%, #f3f4f6);
  --sl-color-accent: #c2410c;
  --sl-color-accent-high: #c2410c;
}
